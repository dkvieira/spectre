19 /*!
20  * \ingroup TestingFrameworkGroup
21  * \brief Given a Map and a CoordinateMapBase, checks that the maps are equal by
22  * downcasting `map_base` and then comparing to `map`. Returns false if the
23  * downcast fails.
24  */
25 template <typename Map>
26 bool are_maps_equal(
27  const Map& map,
28  const CoordinateMapBase<Frame::Logical, Frame::Inertial, Map::dim>&
29  map_base) {
30  const auto* map_derived = dynamic_cast<const Map*>(&map_base);
31  return map_derived == nullptr ? false : (*map_derived == map);
32 }

34 /// \ingroup TestingFrameworkGroup
35 /// \brief Given two coordinate maps (but not their types), check that the maps
36 /// are equal by evaluating them at a random set of points.
37 template <typename SourceFrame, typename TargetFrame, size_t VolumeDim>
38 void check_if_maps_are_equal(
39  const CoordinateMapBase<SourceFrame, TargetFrame, VolumeDim>& map_one,
40  const CoordinateMapBase<SourceFrame, TargetFrame, VolumeDim>& map_two) {
41  std::random_device rd;
42  std::mt19937 gen(rd());
43  std::uniform_real_distribution<> real_dis(-1, 1);
44 
45  for (size_t n = 0; n < 10; ++n) {
46  tnsr::I<double, VolumeDim, SourceFrame> source_point{};
47  for (size_t d = 0; d < VolumeDim; ++d) {
48  source_point.get(d) = real_dis(gen);
49  }
50  CAPTURE_PRECISE(source_point);
51  CHECK_ITERABLE_APPROX(map_one(source_point), map_two(source_point));
52  }
53 }

55 /*!
56  * \ingroup TestingFrameworkGroup
57  * \brief Given a Map `map`, checks that the jacobian gives expected results
58  * when compared to the numerical derivative in each direction.
59  */
60 template <typename Map>
61 void test_jacobian(const Map& map,
62  const std::array<double, Map::dim>& test_point) {
63  // Our default approx value is too stringent for this test
64  Approx local_approx = Approx::custom().epsilon(1e-10).scale(1.0);
65  const double dx = 1e-4;
66  const auto jacobian = map.jacobian(test_point);
67  for (size_t i = 0; i < Map::dim; ++i) {
68  const auto numerical_deriv_i = numerical_derivative(map, test_point, i, dx);
69  for (size_t j = 0; j < Map::dim; ++j) {
70  INFO("i: " << i << " j: " << j);
71  CHECK(jacobian.get(j, i) == local_approx(gsl::at(numerical_deriv_i, j)));
72  }
73  }
74 }

76 /*!
77  * \ingroup TestingFrameworkGroup
78  * \brief Given a Map `map`, checks that the inverse jacobian and jacobian
79  * multiply together to produce the identity matrix
80  */
81 template <typename Map>
82 void test_inv_jacobian(const Map& map,
83  const std::array<double, Map::dim>& test_point) {
84  const auto jacobian = map.jacobian(test_point);
85  const auto inv_jacobian = map.inv_jacobian(test_point);
86 
87  const auto expected_identity = [&jacobian, &inv_jacobian]() {
88  std::array<std::array<double, Map::dim>, Map::dim> identity{};
89  for (size_t i = 0; i < Map::dim; ++i) {
90  for (size_t j = 0; j < Map::dim; ++j) {
91  gsl::at(gsl::at(identity, i), j) = 0.;
92  for (size_t k = 0; k < Map::dim; ++k) {
93  gsl::at(gsl::at(identity, i), j) +=
94  jacobian.get(i, k) * inv_jacobian.get(k, j);
95  }
96  }
97  }
98  return identity;
99  }();
100 
101  for (size_t i = 0; i < Map::dim; ++i) {
102  for (size_t j = 0; j < Map::dim; ++j) {
103  CHECK(gsl::at(gsl::at(expected_identity, i), j) ==
104  approx(i == j ? 1. : 0.));
105  }
106  }
107 }

109 /*!
110  * \ingroup TestingFrameworkGroup
111  * \brief Checks that the CoordinateMap `map` functions as expected when used as
112  * the template parameter to the `CoordinateMap` type.
113  */
114 template <typename Map, typename... Args>
115 void test_coordinate_map_implementation(const Map& map) {
116  const auto coord_map = make_coordinate_map<Frame::Logical, Frame::Grid>(map);
117  std::random_device rd;
118  std::mt19937 gen(rd());
119  std::uniform_real_distribution<> real_dis(-1, 1);
120 
121  const auto test_point = [&gen, &real_dis] {
122  std::array<double, Map::dim> p{};
123  for (size_t i = 0; i < Map::dim; ++i) {
124  gsl::at(p, i) = real_dis(gen);
125  }
126  return p;
127  }();
128 
129  for (size_t i = 0; i < Map::dim; ++i) {
130  CAPTURE_PRECISE(gsl::at(test_point, i));
131  }
132 
133  const auto test_point_tensor = [&test_point]() {
134  tnsr::I<double, Map::dim, Frame::Logical> point_as_tensor{};
135  for (size_t i = 0; i < Map::dim; ++i) {
136  point_as_tensor.get(i) = gsl::at(test_point, i);
137  }
138  return point_as_tensor;
139  }();
140 
141  for (size_t i = 0; i < Map::dim; ++i) {
142  CHECK(coord_map(test_point_tensor).get(i) ==
143  approx(gsl::at(map(test_point), i)));
144  for (size_t j = 0; j < Map::dim; ++j) {
145  CHECK(coord_map.jacobian(test_point_tensor).get(i, j) ==
146  map.jacobian(test_point).get(i, j));
147  CHECK(coord_map.inv_jacobian(test_point_tensor).get(i, j) ==
148  map.inv_jacobian(test_point).get(i, j));
149  }
150  }
151 }

153 /*!
154  * \ingroup TestingFrameworkGroup
155  * \brief Checks that the CoordinateMap `map` functions as expected when used
156  * with different argument types.
157  */
158 template <bool CheckInverse = true, typename Map, typename... Args>
159 void test_coordinate_map_argument_types(
160  const Map& map, const std::array<double, Map::dim>& test_point) {
161  const auto make_array_data_vector = [](const auto& double_array) {
162  std::array<DataVector, Map::dim> result;
163  std::transform(double_array.begin(), double_array.end(), result.begin(),
164  [](const double x) { return DataVector{x, x}; });
165  return result;
166  };
167  const auto make_tensor_data_vector = [](const auto& double_tensor) {
168  using Arg = std::decay_t<decltype(double_tensor)>;
169  Tensor<DataVector, typename Arg::symmetry, typename Arg::index_list> result;
170  std::transform(double_tensor.begin(), double_tensor.end(), result.begin(),
171  [](const double x) { return DataVector{x, x}; });
172  return result;
173  };
174  const auto add_reference_wrapper = [](const auto& unwrapped_array) {
175  using Arg = std::decay_t<decltype(unwrapped_array)>;
176  return make_array<std::reference_wrapper<const typename Arg::value_type>,
177  Map::dim>(unwrapped_array);
178  };
179 
180  const auto mapped_point = map(test_point);
181  CHECK_ITERABLE_APPROX(map(add_reference_wrapper(test_point)), mapped_point);
182  CHECK_ITERABLE_APPROX(map(make_array_data_vector(test_point)),
183  make_array_data_vector(mapped_point));
184  CHECK_ITERABLE_APPROX(
185  map(add_reference_wrapper(make_array_data_vector(test_point))),
186  make_array_data_vector(mapped_point));
187 
188  if (CheckInverse) {
189  const auto expected = map.inverse(mapped_point);
190  CHECK_ITERABLE_APPROX(map.inverse(add_reference_wrapper(mapped_point)),
191  expected);
192  CHECK_ITERABLE_APPROX(map.inverse(make_array_data_vector(mapped_point)),
193  make_array_data_vector(expected));
194  CHECK_ITERABLE_APPROX(map.inverse(add_reference_wrapper(
195  make_array_data_vector(mapped_point))),
196  make_array_data_vector(expected));
197  }
198 
199  {
200  const auto expected = map.jacobian(test_point);
201  CHECK_ITERABLE_APPROX(map.jacobian(add_reference_wrapper(test_point)),
202  expected);
203  CHECK_ITERABLE_APPROX(map.jacobian(make_array_data_vector(test_point)),
204  make_tensor_data_vector(expected));
205  CHECK_ITERABLE_APPROX(
206  map.jacobian(add_reference_wrapper(make_array_data_vector(test_point))),
207  make_tensor_data_vector(expected));
208  }
209 
210  {
211  const auto expected = map.inv_jacobian(test_point);
212  CHECK_ITERABLE_APPROX(map.inv_jacobian(add_reference_wrapper(test_point)),
213  expected);
214  CHECK_ITERABLE_APPROX(map.inv_jacobian(make_array_data_vector(test_point)),
215  make_tensor_data_vector(expected));
216  CHECK_ITERABLE_APPROX(map.inv_jacobian(add_reference_wrapper(
217  make_array_data_vector(test_point))),
218  make_tensor_data_vector(expected));
219  }
220 }
