22 /*!
23  * \ingroup EinsteinSolutionsGroup
24  * \brief Kerr black hole in Kerr-Schild coordinates
25  *
26  * \details
27  * The metric is \f$g_{\mu\nu} = \eta_{\mu\nu} + 2 H l_\mu l_\nu\f$,
28  * where \f$\eta_{\mu\nu}\f$ is the Minkowski metric, \f$H\f$ is a scalar
29  * function, and \f$l_\mu\f$ is the outgoing null vector.
30  * \f$H\f$ and \f$l_\mu\f$ are known functions of the coordinates
31  * and of the mass and spin vector.
32  *
33  * The following are input file options that can be specified:
34  * - Mass (default: 1.)
35  * - Center (default: {0,0,0})
36  * - Spin (default: {0,0,0})
37  *
38  * ## Kerr-Schild Coordinates
39  *
40  *
41  * A Kerr-Schild coordinate system is defined by
42  * \f{equation}{
43  * g_{\mu\nu} \equiv \eta_{\mu\nu} + 2 H l_\mu l_\nu,
44  * \f}
45  * where \f$H\f$ is a scalar function of the coordinates, \f$\eta_{\mu\nu}\f$
46  * is the Minkowski metric, and \f$l^\mu\f$ is a null vector. Note that the
47  * form of the metric along with the nullness of \f$l^\mu\f$ allows one to
48  * raise and lower indices of \f$l^\mu\f$ using \f$\eta_{\mu\nu}\f$, and
49  * that \f$l^t l^t = l_t l_t = l^i l_i\f$.
50  * Note also that
51  * \f{equation}{
52  * g^{\mu\nu} \equiv \eta^{\mu\nu} - 2 H l^\mu l^\nu,
53  * \f}
54  * and that \f$\sqrt{-g}=1\f$.
55  * Also, \f$l_\mu\f$ is a geodesic with respect to both the physical metric
56  * and the Minkowski metric:
57  * \f{equation}{
58  * l^\mu \partial_\mu l_\nu = l^\mu\nabla_\mu l_\nu = 0.
59  * \f}
60  *
61  *
62  * The corresponding 3+1 quantities are
63  * \f{eqnarray}{
64  * g_{i j} &=& \delta_{i j} + 2 H l_i l_j,\\
65  * g^{i j} &=& \delta^{i j} - {2 H l^i l^j \over 1+2H l^t l^t},\\
66  * {\rm det} g_{i j}&=& 1+2H l^t l^t,\\
67  * \beta^i &=& - {2 H l^t l^i \over 1+2H l^t l^t},\\
68  * N &=& \left(1+2 H l^t l^t\right)^{-1/2},\quad\hbox{(lapse)}\\
69  * \alpha &=& \left(1+2 H l^t l^t\right)^{-1},
70  * \quad\hbox{(densitized lapse)}\\
71  * K_{i j} &=& - \left(1+2 H l^t l^t\right)^{1/2}
72  * \left[l_i l_j \partial_{t} H + 2 H l_{(i} \partial_{t} l_{j)}\right]
73  * \nonumber \\
74  * &&-2\left(1+2 H l^t l^t\right)^{-1/2}
75  * \left[H l^t \partial_{(i}l_{j)} + H l_{(i}\partial_{j)}l^t
76  * + l^t l_{(i}\partial_{j)} H + 2H^2 l^t l_{(i} l^k\partial_{k}l_{j)}
77  * + H l^t l_i l_j l^k \partial_{k} H\right],\\
78  * \partial_{k}g_{i j}&=& 2 l_i l_j\partial_{k} H +
79  * 4 H l_{(i} \partial_{k}l_{j)},\\
80  * \partial_{k}N &=& -\left(1+2 H l^t l^t\right)^{-3/2}
81  * \left(l^tl^t\partial_{k}H+2Hl^t\partial_{k}l^t\right),\\
82  * \partial_{k}\beta^i &=& - 2\left(1+2H l^t l^t\right)^{-1}
83  * \left(l^tl^i\partial_{k}H+Hl^t\partial_{k}l^i+Hl^i\partial_{k}l^t\right)
84  * + 4 H l^t l^i \left(1+2H l^t l^t\right)^{-2}
85  * \left(l^tl^t\partial_{k}H+2Hl^t\partial_{k}l^t\right),\\
86  * \Gamma^k{}_{i j}&=& -\delta^{k m}\left(l_i l_j \partial_{m}H
87  * + 2l_{(i} \partial_{m}l_{j)} \right)
88  * + 2 H l_{(i}\partial_{j)} l^k
89  * \nonumber \\
90  * &&+\left(1+2 H l^t l^t\right)^{-1}
91  * \left[2 l^k l_{(i}\partial_{j)} H
92  * +2 H l_i l_j l^k l^m \partial_{m}H
93  * +2 H l^k \partial_{(i}l_{j)}
94  * +4 H^2 l^k l_{(i} (l^m \partial_{m}l_{j)}
95  * -\partial_{j)} l^t)
96  * \right].
97  * \f}
98  * Note that \f$l^i\f$ is **not** equal to \f$g^{i j} l_j\f$; it is equal
99  * to \f${}^{(4)}g^{i \mu} l_\mu\f$.
100  *
101  * ## Kerr Spacetime
102  *
103  * ### Spin in the z direction
104  *
105  * Assume Cartesian coordinates \f$(t,x,y,z)\f$. Then for stationary Kerr
106  * spacetime with mass \f$M\f$ and angular momentum \f$a M\f$
107  * in the \f$z\f$ direction,
108  * \f{eqnarray}{
109  * H &=& {M r^3 \over r^4 + a^2 z^2},\\
110  * l_\mu &=&
111  * \left(1,{rx+ay\over r^2+a^2},{ry-ax\over r^2+a^2},{z\over r}\right),
112  * \f}
113  * where \f$r\f$ is defined by
114  * \f{equation}{
115  * \label{eq:rdefinition1}
116  * {x^2+y^2\over a^2+r^2} + {z^2\over r^2} = 1,
117  * \f}
118  * or equivalently,
119  * \f{equation}{
120  * r^2 = {1\over 2}(x^2 + y^2 + z^2 - a^2)
121  * + \left({1\over 4}(x^2 + y^2 + z^2 - a^2)^2 + a^2 z^2\right)^{1/2}.
122  * \f}
123  *
124  * Possibly useful formula:
125  * \f{equation}{
126  * \partial_{i} r = {x_i + z \delta_{i z} \displaystyle {a^2\over r^2} \over
127  * 2 r\left(1 - \displaystyle {x^2 + y^2 + z^2 - a^2\over 2 r^2}\right)}.
128  * \f}
129  *
130  * ### Spin in an arbitrary direction
131  *
132  * For arbitrary spin direction, let \f$\vec{x}\equiv (x,y,z)\f$ and
133  * \f$\vec{a}\f$ be a flat-space three-vector with magnitude-squared
134  * (\f$\delta_{ij}\f$ norm) equal to \f$a^2\f$.
135  * Then the Kerr-Schild quantities for Kerr spacetime are:
136  * \f{eqnarray}{
137  * H &=& {M r^3 \over r^4 + (\vec{a}\cdot\vec{x})^2},\\
138  * \vec{l} &=& {r\vec{x}-\vec{a}\times\vec{x}+(\vec{a}\cdot\vec{x})\vec{a}/r
139  * \over r^2+a^2 },\\
140  * l_t &=& 1,\\
141  * \label{eq:rdefinition2}
142  * r^2 &=& {1\over 2}(\vec{x}\cdot\vec{x}-a^2)
143  * + \left({1\over 4}(\vec{x}\cdot\vec{x}-a^2)^2
144  * + (\vec{a}\cdot\vec{x})^2\right)^{1/2},
145  * \f}
146  * where \f$\vec{l}\equiv (l_x,l_y,l_z)\f$, and
147  * all dot and cross products are evaluated as flat-space 3-vector operations.
148  *
149  * Possibly useful formulae:
150  * \f{equation}{
151  * \partial_{i} r = {x_i + (\vec{a}\cdot\vec{x})a_i/r^2 \over
152  * 2 r\left(1 - \displaystyle {\vec{x}\cdot\vec{x}-a^2\over 2 r^2}\right)},
153  * \f}
154  * \f{equation}{
155  * {\partial_{i} H \over H} =
156  * {3\partial_{i}r\over r} - {4 r^3 \partial_{i}r +
157  * 2(\vec{a}\cdot\vec{x})\vec{a}
158  * \over r^4 + (\vec{a}\cdot\vec{x})^2},
159  * \f}
160  * \f{equation}{
161  * (r^2+a^2)\partial_{j} l_i =
162  * (x_i-2 r l_i-(\vec{a}\cdot\vec{x})a_i/r^2)\partial_{j}r
163  * + r\delta_{ij} + a_i a_j/r + \epsilon^{ijk} a_k.
164  * \f}
165  *
166  * ## Cartesian and Spherical Coordinates for Kerr
167  *
168  * The Kerr-Schild coordinates are defined in terms of the Cartesian
169  * coordinates \f$(x,y,z)\f$. If one wishes to express Kerr-Schild
170  * coordinates in terms of the spherical polar coordinates
171  * \f$(\tilde{r},\theta,\phi)\f$ then one can make the obvious and
172  * usual transformation
173  * \f{equation}{
174  * \label{eq:sphertocartsimple}
175  * x=\tilde{r}\sin\theta\cos\phi,\quad
176  * y=\tilde{r}\sin\theta\sin\phi,\quad
177  * z=\tilde{r}\cos\theta.
178  * \f}
179  *
180  * This is simple, and has the advantage that in this coordinate system
181  * for \f$M\to0\f$, Kerr spacetime becomes Minkowski space in spherical
182  * coordinates \f$(\tilde{r},\theta,\phi)\f$. However, the disadvantage is
183  * that the horizon of a Kerr hole is **not** located at constant
184  * \f$\tilde{r}\f$, but is located instead at constant \f$r\f$,
185  * where \f$r\f$ is the radial
186  * Boyer-Lindquist coordinate defined in (\f$\ref{eq:rdefinition2}\f$).
187  *
188  * For spin in the \f$z\f$ direction, one could use the transformation
189  * \f{equation}{
190  * x=\sqrt{r^2+a^2}\sin\theta\cos\phi,\quad
191  * y=\sqrt{r^2+a^2}\sin\theta\sin\phi,\quad
192  * z=r\cos\theta.
193  * \f}
194  * In this case, for \f$M\to0\f$, Kerr spacetime becomes Minkowski space in
195  * spheroidal coordinates, but now the horizon is on a constant-coordinate
196  * surface.
197  *
198  * Right now we use (\f$\ref{eq:sphertocartsimple}\f$), but we may
199  * wish to use the other transformation in the future.
200  */
