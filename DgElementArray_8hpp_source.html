64  for (const auto& block : domain.blocks()) {

67  const std::vector<ElementId<volume_dim>> element_ids =
68  initial_element_ids(block.id(), initial_ref_levs);
69  int which_proc = 0;
70  const int number_of_procs = Parallel::number_of_procs();
71  for (size_t i = 0; i < element_ids.size(); ++i) {
72  dg_element_array(ElementIndex<volume_dim>(element_ids[i]))
73  .insert(global_cache, which_proc);
74  which_proc = which_proc + 1 == number_of_procs ? 0 : which_proc + 1;
75  }
76  }
