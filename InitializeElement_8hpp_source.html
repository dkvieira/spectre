35 /// \brief Initialize a dG element with analytic initial data
36 ///
37 /// Uses:
38 /// - ConstGlobalCache:
39 /// * A tag deriving off of Cache::AnalyticSolutionBase
40 /// * CacheTags::TimeStepper
41 ///
42 /// DataBox changes:
43 /// - Adds:
44 /// * Tags::TimeId
45 /// * Tags::Time
46 /// * Tags::TimeStep
47 /// * Tags::LogicalCoordinates<Dim>
48 /// * Tags::Extents<Dim>
49 /// * Tags::Element<Dim>
50 /// * Tags::ElementMap<Dim>
51 /// * System::variables_tag
52 /// * Tags::HistoryEvolvedVariables<System::variables_tag,
53 /// db::add_tag_prefix<Tags::dt, System::variables_tag>>
54 /// * Tags::Coordinates<Tags::ElementMap<Dim>,
55 /// Tags::LogicalCoordinates<Dim>>
56 /// * Tags::InverseJacobian<Tags::ElementMap<Dim>,
57 /// Tags::LogicalCoordinates<Dim>>

61 /// Tags::LogicalCoordinates<Dim>>>

65 /// - Modifies: nothing

90  template <typename... InboxTags, typename Metavariables, typename ActionList,
91  typename ParallelComponent>
92  static auto apply(const db::DataBox<tmpl::list<>>& /*box*/,
93  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
94  const Parallel::ConstGlobalCache<Metavariables>& cache,
95  const ElementIndex<Dim>& array_index,
96  const ActionList /*meta*/,
97  const ParallelComponent* const /*meta*/,
98  std::vector<std::array<size_t, Dim>> initial_extents,
99  Domain<Dim, Frame::Inertial> domain, Time initial_time,
100  TimeDelta initial_dt) noexcept {
101  using solution_tag = CacheTags::AnalyticSolutionBase;
102  using variables_tags =
103  typename Metavariables::system::variables_tag::tags_list;
104 
105  ElementId<Dim> element_id{array_index};
106 
107  const auto& my_block = domain.blocks()[element_id.block_id()];
108 
109  ElementMap<Dim, Frame::Inertial> map{element_id,
110  my_block.coordinate_map().get_clone()};
111 
112  Element<Dim> element = create_initial_element(element_id, my_block);
113 
114  ::Index<Dim> mesh{initial_extents[element_id.block_id()]};
115  const auto num_grid_points = mesh.product();
116  auto logical_coords = logical_coordinates(mesh);
117  auto inertial_coords = map(logical_coords);
118 
119  // Set initial data from analytic solution
120  auto vars = Parallel::get<solution_tag>(cache).evolution_variables(
121  inertial_coords, initial_time.value());
122 
123  typename Tags::HistoryEvolvedVariables<
124  Tags::Variables<variables_tags>,
125  Tags::dt<Tags::Variables<db::wrap_tags_in<Tags::dt, variables_tags>>>>::
126  type history_dt_vars;
127  const auto& time_stepper = Parallel::get<CacheTags::TimeStepper>(cache);
128  if (not time_stepper.is_self_starting()) {
129  // We currently just put initial points at past slab boundaries.
130  Time past_t = initial_time;
131  TimeDelta past_dt = initial_dt;
132  for (size_t i = time_stepper.number_of_past_steps(); i > 0; --i) {
133  past_dt = past_dt.with_slab(past_dt.slab().advance_towards(-past_dt));
134  past_t -= past_dt;
135 
136  history_dt_vars.insert_initial(
137  past_t,
138  Parallel::get<solution_tag>(cache).evolution_variables(
139  inertial_coords, past_t.value()),
140  Parallel::get<solution_tag>(cache).dt_evolution_variables(
141  inertial_coords, past_t.value()));
142  }
143  }
144 
145  // Set up initial time
146  TimeId time_id{};
147  time_id.time = initial_time;
148 
149  db::compute_databox_type<return_tag_list<typename Metavariables::system>>
150  outbox = db::create<
151  db::get_items<return_tag_list<typename Metavariables::system>>,
152  db::get_compute_items<
153  return_tag_list<typename Metavariables::system>>>(
154  time_id, initial_dt, std::move(mesh), std::move(element),
155  std::move(map), std::move(vars), std::move(history_dt_vars),
156  Variables<db::wrap_tags_in<Tags::dt, variables_tags>>{
157  num_grid_points, 0.0});
158 
159  return std::make_tuple(std::move(outbox));
160  }
161 };
