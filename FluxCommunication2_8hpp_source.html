70  typename... ArgumentTags, typename... Args>
71  static void apply_normal_dot_numerical_flux(
72  const gsl::not_null<Variables<tmpl::list<NumericalFluxTags...>>*>

82  public:
83  template <typename DbTags, typename... InboxTags, typename ArrayIndex,
84  typename ActionList, typename ParallelComponent>
85  static auto apply(db::DataBox<DbTags>& box,
86  tuples::TaggedTuple<InboxTags...>& inboxes,
87  const Parallel::ConstGlobalCache<Metavariables>& cache,
88  const ArrayIndex& /*array_index*/,
89  const ActionList /*meta*/,
90  const ParallelComponent* const /*meta*/) noexcept {
91  using variables_tag = typename system::variables_tag;
92  using dt_variables_tag = db::add_tag_prefix<Tags::dt, variables_tag>;
93 
94  using normal_dot_flux_tag =
95  db::add_tag_prefix<Tags::NormalDotFlux, variables_tag>;
96  using normal_dot_numerical_flux_tag =
97  db::add_tag_prefix<Tags::NormalDotNumericalFlux, variables_tag>;
98  using mortars_normal_dot_fluxes_tag =
99  DgActions_detail::mortars_tag<volume_dim, normal_dot_flux_tag>;
100  using mortars_packaged_data_tag = DgActions_detail::mortars_tag<
101  volume_dim,
102  Tags::Variables<typename Metavariables::normal_dot_numerical_flux::
103  type::package_tags>>;
104 
105  const auto& normal_dot_numerical_flux_computer =
106  get<typename Metavariables::normal_dot_numerical_flux>(cache);
107  const auto& extents = db::get<Tags::Extents<volume_dim>>(box);
108  const auto& local_mortar_data = db::get<mortars_packaged_data_tag>(box);
109  const auto& local_normal_dot_flux =
110  db::get<mortars_normal_dot_fluxes_tag>(box);
111 
112  const auto& element = db::get<Tags::Element<volume_dim>>(box);
113 
114  auto& inbox = tuples::get<FluxesTag>(inboxes);
115  const auto& time_id = db::get<Tags::TimeId>(box);
116  auto remote_data = std::move(inbox[time_id]);
117  inbox.erase(time_id);
118 
119  for (const auto& direction_neighbors : element.neighbors()) {
120  const auto& direction = direction_neighbors.first;
121  const size_t dimension = direction.dimension();
122  const auto& neighbors_in_direction = direction_neighbors.second;
123  ASSERT(neighbors_in_direction.size() == 1,
124  "Complex mortars unimplemented");
125  const auto& neighbor = *neighbors_in_direction.begin();
126  const auto neighbor_data_it =
127  remote_data.find(std::make_pair(direction, neighbor));
128  ASSERT(remote_data.end() != neighbor_data_it,
129  "Attempting to access neighbor flux data that does not exist. "
130  "The neighbor attempting to be accessed is: "
131  << neighbor << " in direction " << direction
132  << ". The known keys are " << keys_of(remote_data) << ".");
133  const PackagedData& self_packaged_data =
134  local_mortar_data.at(std::make_pair(direction, neighbor));
135  const PackagedData& neighbor_packaged_data = neighbor_data_it->second;
136 
137  // All of this needs to be fixed for hp-adaptivity to handle
138  // projections correctly
139  const auto& face_normal =
140  db::get<Tags::UnnormalizedFaceNormal<volume_dim>>(box).at(direction);
141 
142  // Compute numerical flux
143  db::item_type<normal_dot_numerical_flux_tag> normal_dot_numerical_fluxes(
144  face_normal.begin()->size(), 0.0);
145  apply_normal_dot_numerical_flux(
146  make_not_null(&normal_dot_numerical_fluxes),
147  normal_dot_numerical_flux_computer, self_packaged_data,
148  neighbor_packaged_data);
149 
150  // Needs fixing for GH/curved
151  db::item_type<dt_variables_tag> lifted_data(dg::lift_flux(
152  local_normal_dot_flux.at(std::make_pair(direction, neighbor)),
153  std::move(normal_dot_numerical_fluxes), extents[dimension],
154  get(magnitude(face_normal))));
155 
156  db::mutate<dt_variables_tag>(box, [
157  &lifted_data, &extents, &dimension, &direction
158  ](db::item_type<dt_variables_tag> & dt_vars) noexcept {
159  add_slice_to_data(
160  make_not_null(&dt_vars), lifted_data, extents, dimension,
161  direction.side() == Side::Lower ? 0 : extents[dimension] - 1);
162  });
163  }
164 
165  return std::make_tuple(
166  db::create_from<db::RemoveTags<mortars_normal_dot_fluxes_tag,
167  mortars_packaged_data_tag>>(
168  std::move(box)));
169  }

171  template <typename DbTags, typename... InboxTags, typename ArrayIndex>
172  static bool is_ready(
173  const db::DataBox<DbTags>& box,
174  const tuples::TaggedTuple<InboxTags...>& inboxes,
175  const Parallel::ConstGlobalCache<Metavariables>& /*cache*/,
176  const ArrayIndex& /*array_index*/) noexcept {
177  const auto& time_id = db::get<Tags::TimeId>(box);
178  const auto& inbox = tuples::get<FluxesTag>(inboxes);
179  auto receives = inbox.find(time_id);
180  const auto number_of_neighbors =
181  db::get<Tags::Element<volume_dim>>(box).number_of_neighbors();

183  receives == inbox.end() ? 0 : receives->second.size();
184  return num_receives == number_of_neighbors;
185  }

204  template <typename Metavariables, typename... NormalDotFluxTags,
205  typename... VariablesTags, size_t Dim, typename Frame,
206  typename... ArgumentTags>
207  static void apply_flux(
208  gsl::not_null<Variables<tmpl::list<NormalDotFluxTags...>>*> boundary_flux,
209  const Variables<tmpl::list<VariablesTags...>>& boundary_variables,
210  const tnsr::i<DataVector, Dim, Frame>& unit_face_normal,
211  tmpl::list<ArgumentTags...> /*meta*/) noexcept {
212  Metavariables::system::normal_dot_fluxes::apply(
213  make_not_null(&get<NormalDotFluxTags>(*boundary_flux))...,
214  get<ArgumentTags>(boundary_variables)..., unit_face_normal);
215  }

217  public:
218  template <typename DbTags, typename... InboxTags, typename Metavariables,
219  typename ArrayIndex, typename ActionList,
220  typename ParallelComponent>
221  static auto apply(db::DataBox<DbTags>& box,
222  tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
223  Parallel::ConstGlobalCache<Metavariables>& cache,
224  const ArrayIndex& /*array_index*/,
225  const ActionList /*meta*/,
226  const ParallelComponent* const /*meta*/) noexcept {
227  using system = typename Metavariables::system;
228  constexpr size_t volume_dim = system::volume_dim;
229  using variables_tag = typename system::variables_tag;
230  using PackagedData = Variables<
231  typename Metavariables::normal_dot_numerical_flux::type::package_tags>;
232 
233  const auto& normal_dot_numerical_flux_computer =
234  get<typename Metavariables::normal_dot_numerical_flux>(cache);
235 
236  auto& receiver_proxy =
237  Parallel::get_parallel_component<ParallelComponent>(cache);
238 
239  const auto& element = db::get<Tags::Element<volume_dim>>(box);
240  const auto& extents = db::get<Tags::Extents<volume_dim>>(box);
241  const auto& time_id = db::get<Tags::TimeId>(box);
242 
243  using normal_dot_flux_tag =
244  db::add_tag_prefix<Tags::NormalDotFlux, variables_tag>;
245  using NormalDotFluxType = db::item_type<normal_dot_flux_tag>;
246 
247  std::unordered_map<
248  std::pair<Direction<volume_dim>, ElementId<volume_dim>>, PackagedData,
249  boost::hash<std::pair<Direction<volume_dim>, ElementId<volume_dim>>>>
250  mortars_packaged_data;
251  std::unordered_map<
252  std::pair<Direction<volume_dim>, ElementId<volume_dim>>,
253  NormalDotFluxType,
254  boost::hash<std::pair<Direction<volume_dim>, ElementId<volume_dim>>>>
255  mortars_normal_dot_fluxes;
256 
257  for (const auto& direction_neighbors : element.neighbors()) {
258  const auto& direction = direction_neighbors.first;
259  const size_t dimension = direction.dimension();
260  const auto& neighbors_in_direction = direction_neighbors.second;
261  ASSERT(neighbors_in_direction.size() == 1,
262  "h-adaptivity is not supported yet.\nDirection: "
263  << direction << "\nDimension: " << dimension
264  << "\nNeighbors:\n"
265  << neighbors_in_direction);
266  const auto& orientation = neighbors_in_direction.orientation();
267  const auto boundary_extents = extents.slice_away(dimension);
268  auto boundary_variables = db::data_on_slice(
269  box, extents, dimension,
270  direction.side() == Side::Lower ? 0 : extents[dimension] - 1,
271  tmpl::remove_duplicates<
272  tmpl::append<typename Metavariables::normal_dot_numerical_flux::
273  type::slice_tags,
274  typename Metavariables::system::normal_dot_fluxes::
275  argument_tags>>{});
276 
277  // Everything in the loop over neighbors needs to be fixed for
278  // hp-adaptivity to handle projections correctly
279  for (const auto& neighbor : neighbors_in_direction) {
280  const auto& face_normal =
281  db::get<Tags::UnnormalizedFaceNormal<volume_dim>>(box).at(
282  direction);
283  DataVector magnitude_of_face_normal = get(magnitude(face_normal));
284  std::decay_t<decltype(face_normal)> unit_face_normal(
285  magnitude_of_face_normal.size(), 0.0);
286  for (size_t d = 0; d < volume_dim; ++d) {
287  unit_face_normal.get(d) =
288  face_normal.get(d) / magnitude_of_face_normal;
289  }
290 
291  // This needs to be updated for conservative systems where we can just
292  // slice the flux from the volume.
293  NormalDotFluxType local_boundary_flux(face_normal.begin()->size(), 0.0);
294  apply_flux<Metavariables>(
295  make_not_null(&local_boundary_flux), boundary_variables,
296  unit_face_normal,
297  typename system::normal_dot_fluxes::argument_tags{});
298 
299  // We compute the parts of the numerical flux that only depend on data
300  // from this side of the mortar now, then package it into a Variables.
301  // We store one copy of the Variables and send another, since we need
302  // the data on both sides of the mortar.
303  PackagedData packaged_data(face_normal.begin()->size(), 0.0);
304  get_packaged_data_from_numerical_flux(
305  make_not_null(&packaged_data), normal_dot_numerical_flux_computer,
306  local_boundary_flux, boundary_variables,
307  typename Metavariables::normal_dot_numerical_flux::type::
308  slice_tags{},
309  unit_face_normal);
310 
311  const auto direction_from_neighbor = orientation(direction.opposite());
312 
313  // orient_variables_on_slice only needs to be done in the case where
314  // the data is oriented differently. This needs to improved later.
315  // Note: avoiding the same-orientation-on-both-sides copy is possible
316  // even with AMR since the quantities are already on the mortar at this
317  // point
318  auto neighbor_packaged_data = orient_variables_on_slice(
319  packaged_data, boundary_extents, dimension, orientation);
320 
321  receiver_proxy[neighbor]
322  .template receive_data<
323  typename ComputeBoundaryFlux<Metavariables>::FluxesTag>(
324  time_id, std::make_pair(std::make_pair(direction_from_neighbor,
325  element.id()),
326  std::move(neighbor_packaged_data)));
327 
328  mortars_packaged_data.emplace(std::make_pair(direction, neighbor),
329  std::move(packaged_data));
330  mortars_normal_dot_fluxes.emplace(std::make_pair(direction, neighbor),
331  std::move(local_boundary_flux));
332  } // loop over neighbors_in_direction
333  } // loop over element.neighbors()
334 
335  using mortars_normal_dot_fluxes_tag =
336  DgActions_detail::mortars_tag<volume_dim, normal_dot_flux_tag>;
337  using mortars_packaged_data_tag = DgActions_detail::mortars_tag<
338  volume_dim,
339  Tags::Variables<typename Metavariables::normal_dot_numerical_flux::
340  type::package_tags>>;
341  return std::make_tuple(
342  db::create_from<db::RemoveTags<>,
343  db::AddTags<mortars_packaged_data_tag,
344  mortars_normal_dot_fluxes_tag>>(
345  box, std::move(mortars_packaged_data),
346  std::move(mortars_normal_dot_fluxes)));
347  }
