103 template <typename InputIter1, typename InputIter2,
104  typename BinaryPred = detail::Equal<>>
105 constexpr bool equal(InputIter1 first1, InputIter1 last1, InputIter2 first2,
106  BinaryPred pred = detail::Equal<>{}) {
107  for (; first1 != last1; ++first1, ++first2) {
108  if (not pred(*first1, *first2)) {
109  return false;
110  }
111  }
112  return true;
113 }

115 template <typename InputIter1, typename InputIter2,
116  typename BinaryPred = std::less_equal<>>
117 constexpr bool lexicographical_compare(InputIter1 first1, InputIter1 last1,
118  InputIter2 first2, InputIter2 last2,
119  BinaryPred pred = std::less_equal<>{}) {
120  for (; first2 != last2; ++first1, ++first2) {
121  if (first1 == last1 or pred(*first1, *first2)) {
122  return true;
123  } else if (pred(*first2, *first1)) {
124  return false;
125  }
126  }
127  return false;
128 }
