16 /*!
17  * \ingroup DomainCreators
18  * \brief A mapping of the logical coordinate axes of a host to the logical
19  * coordinate axes of a neighbor of the host.
20  * \usage Given a `size_t dimension`, a `Direction`, or a `SegmentId` of the
21  * host, an `OrientationMap` will give the corresponding value in the neighbor.
22  * \tparam VolumeDim the dimension of the blocks.
23  */
24 template <size_t VolumeDim>
25 class OrientationMap {
26  public:
27  /// The default orientation is the identity map on directions.
28  /// The bool `is_aligned_` is correspondingly set to `true`.
29  OrientationMap();
30  explicit OrientationMap(
31  std::array<Direction<VolumeDim>, VolumeDim> mapped_directions);
32  OrientationMap(
33  const std::array<Direction<VolumeDim>, VolumeDim>& directions_in_host,
34  const std::array<Direction<VolumeDim>, VolumeDim>&
35  directions_in_neighbor);
36  ~OrientationMap() = default;
37  OrientationMap(const OrientationMap&) = default;
38  OrientationMap& operator=(const OrientationMap&) = default;
39  OrientationMap(OrientationMap&& /*rhs*/) noexcept = default;
40  OrientationMap& operator=(OrientationMap&& /*rhs*/) noexcept = default;
41 
42  /// True when mapped(Direction) == Direction
43  bool is_aligned() const noexcept { return is_aligned_; }
44 
45  /// The corresponding dimension in the neighbor.
46  size_t operator()(const size_t dim) const noexcept {
47  return gsl::at(mapped_directions_, dim).dimension();
48  }
49 
50  /// The corresponding direction in the neighbor.
51  Direction<VolumeDim> operator()(const Direction<VolumeDim>& direction) const {
52  return direction.side() == Side::Upper
53  ? gsl::at(mapped_directions_, direction.dimension())
54  : gsl::at(mapped_directions_, direction.dimension()).opposite();
55  }
56 
57  /// The corresponding SegmentIds in the neighbor.
58  std::array<SegmentId, VolumeDim> operator()(
59  const std::array<SegmentId, VolumeDim>& segmentIds) const;
60 
61  /// The corresponding Orientation of the host in the frame of the neighbor.
62  OrientationMap<VolumeDim> inverse_map() const noexcept;
63 
64  /// Serialization for Charm++
65  void pup(PUP::er& p); // NOLINT
66 
67  private:
68  friend bool operator==(const OrientationMap& lhs,
69  const OrientationMap& rhs) noexcept {
70  return (lhs.mapped_directions_ == rhs.mapped_directions_ and
71  lhs.is_aligned_ == rhs.is_aligned_);
72  }
73 
74  std::array<Direction<VolumeDim>, VolumeDim> mapped_directions_;
75  bool is_aligned_ = true;
76 };
